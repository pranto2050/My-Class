



C++-এ কনস্ট্রাক্টর হলো একটি বিশেষ মেম্বার ফাংশন যা কোনো ক্লাসের অবজেক্ট তৈরি হওয়ার সাথে সাথেই স্বয়ংক্রিয়ভাবে কল হয়। 
এর প্রধান কাজ হলো ক্লাসের মেম্বার ভেরিয়েবলগুলিকে প্রাথমিক মান (initialize) দেওয়া। 
কনস্ট্রাক্টরের নাম অবশ্যই ক্লাসের নামের সাথে মিল রাখতে হবে এবং এর কোনো রিটার্ন টাইপ থাকে না (এমনকি void ও নয়)।






কনস্ট্রাক্টরের বৈশিষ্ট্য:
    1. ক্লাসের নামের সাথে মিল: কনস্ট্রাক্টরের নাম অবশ্যই ক্লাসের নামের মতো হতে হবে।
    2. রিটার্ন টাইপ নেই: কনস্ট্রাক্টর কোনো মান রিটার্ন করে না।
    3. স্বয়ংক্রিয়ভাবে কল হয়: অবজেক্ট তৈরি হওয়ার সময় এটি স্বয়ংক্রিয়ভাবে চালু হয়।
    4. ওভারলোড করা যায়: একাধিক কনস্ট্রাক্টর ভিন্ন প্যারামিটার দিয়ে তৈরি করা সম্ভব।
    5. ডিফল্ট কনস্ট্রাক্টর: যদি কোনো কনস্ট্রাক্টর না দেওয়া হয়, তবে C++ একটি ডিফল্ট কনস্ট্রাক্টর সরবরাহ করে।



কনস্ট্রাক্টরের প্রকারভেদ: 

    1. ডিফল্ট কনস্ট্রাক্টর (Default Constructor)
        কোনো প্যারামিটার ছাড়াই কাজ করে।
class Example {
public:
    Example() { 
        cout << "ডিফল্ট কনস্ট্রাক্টর কল হয়েছে!" << endl; 
    }
};








2. প্যারামিটারাইজড কনস্ট্রাক্টর (Parameterized Constructor)
    প্যারামিটার নিয়ে কাজ করে এবং ক্লাসের মেম্বার ভেরিয়েবলগুলোকে নির্দিষ্ট মান দেয়।

class Example {
private:
    int x, y;
public:
    Example(int a, int b) { 
        x = a; 
        y = b; 
        cout << "প্যারামিটারাইজড কনস্ট্রাক্টর কল হয়েছে!" << endl;
    }
};





3. কপি কনস্ট্রাক্টর (Copy Constructor)
    একটি অবজেক্ট তৈরি করার সময় অন্য অবজেক্টের ডেটা কপি করতে ব্যবহার করা হয়।

class Example {
private:
    int x;
public:
    Example(int a) { 
        x = a; 
        }
    Example(const Example &obj) { 
        x = obj.x; 
        cout << "কপি কনস্ট্রাক্টর কল হয়েছে!" << endl; 
    }
};






একটি পুরো কোড ------- 

#include <iostream>
using namespace std;

class Person {
private:
    string name;
    int age;

public:
    // প্যারামিটারাইজড কনস্ট্রাক্টর
    Person(string n, int a) {
        name = n;
        age = a;
    }

    // কপি কনস্ট্রাক্টর
    Person(const Person &p) {
        name = p.name;
        age = p.age;
        cout << "কপি কনস্ট্রাক্টর কল হয়েছে!" << endl;
    }

    void display() {
        cout << "নাম: " << name << ", বয়স: " << age << endl;
    }
};

int main() {
    Person p1("Alice", 30); // প্যারামিটারাইজড কনস্ট্রাক্টর
    Person p2 = p1;         // কপি কনস্ট্রাক্টর কল হবে

    cout << "p1-এর তথ্য:" << endl;
    p1.display();

    cout << "p2-এর তথ্য:" << endl;
    p2.display();

    return 0;
}
